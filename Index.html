import requests
import json
from datetime import datetime

class WeatherAPIClient:
    """
    A weather API client for fetching current weather data.
    Generated by VoxAbility AI via voice command:
    'Create a Python class to fetch weather data from OpenWeatherMap API'
    """
    
    def __init__(self, api_key):
        self.base_url = "https://api.openweathermap.org/data/2.5/"
        self.api_key = api_key
    
    def get_current_weather(self, city, country_code=None):
        """Fetch current weather for a given city"""
        location = city
        if country_code:
            location += "," + country_code
        
        params = {
            "q": location,
            "appid": self.api_key,
            "units": "metric"
        }
        
        try:
            response = requests.get(
                self.base_url + "weather", 
                params=params
            )
            response.raise_for_status()
            
            return self._format_weather_data(response.json())
            
        except requests.exceptions.RequestException as e:
            return {"error": f"API request failed: {str(e)}"}
    
    def _format_weather_data(self, data):
        """Format raw API response into user-friendly structure"""
        return {
            "location": data["name"],
            "temperature": data["main"]["temp"],
            "feels_like": data["main"]["feels_like"],
            "humidity": data["main"]["humidity"],
            "description": data["weather"][0]["description"],
            "wind_speed": data["wind"]["speed"],
            "timestamp": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        }
    
    def get_weather_alert(self, city):
        """Check for severe weather conditions"""
        weather = self.get_current_weather(city)
        
        if "error" not in weather:
            alerts = []
            
            if weather["wind_speed"] > 20:
                alerts.append("High wind warning")
            
            if weather["temperature"] < 0:
                alerts.append("Freezing temperature alert")
            
            return {
                "weather": weather,
                "alerts": alerts
            }
        
        return weather
